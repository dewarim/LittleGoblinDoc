001package de.dewarim.goblin;
002
003/**
004 *
005 */
006public enum ComponentType {
007
008    /**
009     * Components of type INPUT will be used up during production.
010     */
011    INPUT,
012
013    /**
014     * Components of type OUTPUT will be created by the production process.
015     */
016    OUTPUT,
017
018    /**
019     * Components of type TOOL are required for a production process, but will
020     * not be consumed by it. Example: an anvil may be used to create a sword.
021     */
022    TOOL
023
024}
