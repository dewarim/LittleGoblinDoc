002
003public enum MeleeStatus {
004
005    /**
006     * Before a Melee starts, the system has to wait for the participants to register.
007     */
008    WAITING,
009
010    /**
011     * A Melee that has been started is in the RUNNING state.
012     */
013    RUNNING,
014
015    /**
016     * After the last (wo)man standing has been determined (or all Players have left),
017     * the Melee is finished.
018     */
019    FINISHED
020
021}
